Below are a few samples of analysis I’ve completed. All analysis was done in the RStudio IDE. Click the Code button above the figure to view the code used to create each graph.

March 2024

This analysis was to explore the top industries and applicants in March 2024. The first graph is a visualization of applicants by school year, including advanced degrees and alumni. The second graph is visualizes the total applicants by gender.

Code 
sy_longer <- march_school_year |> 
  pivot_longer(
    cols = alumni:sophomore,
    names_to = "school_year",
    values_to = "applicants",
    values_drop_na = TRUE
  )

sy_longer |> 
  mutate(school_year = school_year |> fct_infreq() |> fct_rev()) |> 
  ggplot(aes(school_year, fill = school_year)) +
  geom_bar(show.legend = FALSE)+
  scale_color_brewer(palette = "Set1")+
  labs(
    x = "",
    y = "",
    title = "Number of Applicants by School Year",
    subtitle = "March 2024"
  )
Code 
march_longer <- march_gender |> 
  pivot_longer(
    cols = female:male,
    names_to = "gender",
    values_to = "applicants",
    values_drop_na = TRUE
  )

march_longer |> 
  drop_na(job_type) |> 
  mutate(job_type = job_type |> fct_rev()) |> 
  ggplot(aes(job_type, fill = gender)) +
  geom_bar(position = "dodge")+
  scale_color_brewer(palette = "Set1")+
  labs(x = "",
       y = "",
       fill = "Gender",
       title = "Applicants by Job Type and Gender", 
       subtitle = "March 2024")

applications <- my_postings 

applications |> 
group_by(industry) |> 
  summarize(total = sum(applications_count)) |> 
  arrange(desc(total)) %>% 
  slice_head(n = 20) %>% 
  ungroup() %>% 
  mutate(industry = fct_reorder(industry, total)) %>% 
ggplot(aes(industry, total, fill = industry)) +
  geom_col() +
  coord_flip() +
  theme(legend.position = "none") +
  labs(x = "Industry",
       y = "Number of applicants",
       title = "Total Student Applicants by Industry",
       subtitle = "November 2024 to February 2025")
